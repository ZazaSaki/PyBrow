729def VerifyerBuilder(SpreadSheet:Spreadsheet)->Tuple:
730 queryLanguageSelector = '"select *"'
731 GeneralCode = ""
732 tableList = []
733 for workSheet in SpreadSheet:
734 if not validate(workSheet):
735 continue
736
737 range = code_title()
738 iteration_code = "'" + workSheet.title.replace("'", "''") + "'!" + range
739 GeneralCode += iteration_code + ","
740
741 tableList.append( "=QUERY({"+ iteration_code + "}," + queryLanguageSelector + ", 0 )")
742
743 GeneralCode = "{" + GeneralCode[0:-1] + "}"
744
745 range = "A1:{col}2".format(col = get_col_by_num(len(tableList)))

748 return [[GeneralCode], tableList], range
